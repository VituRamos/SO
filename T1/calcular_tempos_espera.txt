#Formatar linha do tempo 
def linha_tempo_calc(linha_tempo):
    tempo = 0
    processo_atual = None
    linha_tempo_mod = []

    for processo in linha_tempo:
        
        if processo == processo_atual:
            tempo += 1
            
        else:
            
            if processo_atual:
                
                linha_tempo_mod += [(processo_atual,tempo)]
            
            processo_atual = processo
            tempo += 1

    if processo_atual:
        linha_tempo_mod += [(processo_atual,tempo)]

    return linha_tempo_mod
